\begin{frame}{Warum crypto?}
	\begin{itemize}
		\item Privatsphäre trotz Totalüberwachung
		\item (Geschäfts-) Geheimnisse bewahren
		\item Ermöglicht andere uns verschlüsst zu erreichen
		\item Cryptonoise
	\end{itemize}

	\note[item]{Everyone: Right to privacy, right to make mistakes, right to keep secrets.}
	\note[item]{If total observation was fine (and no places to hide should be left)
	than this implies that everybody's living room should be subject to surveillance as well. Probably not what we want.}
	\note[item]{Observation kills free speech, because people fear.}
	\note[item]{Upside of using crypto: Less exposure to surveillance of people around you}
	\note[item]{Upside2: If you want to others to reach you cryptoed, then
	you need to do the first step.} 
	\note[item]{Obligation to do the first step.}
	\note[item]{Downside: Initially one is more interesting --- if you are using crypto something changes on the surveillance radar}
	\note[item]{But you are surveilled anyway, so \dots}
	\note[item]{Cryptonoise helps both terrorists and activists to hide.}
\end{frame}
